<script>
const storeDomain = 'presidentecartel.myshopify.com';
const storefrontToken = 'cca64ccf85882f759c632c32fe2663dd';

const productsContainer = document.getElementById('products');
const entrance = document.getElementById('entrance');
const store = document.getElementById('store');
const cartContainer = document.getElementById('cart-items');
const cartTotalEl = document.getElementById('cart-total');
let cart = [];

document.getElementById('enterBtn').addEventListener('click', () => {
  entrance.style.transition = "opacity 1s";
  entrance.style.opacity = 0;
  setTimeout(() => {
    entrance.style.display = 'none';
    store.style.display = 'block';
    loadProducts();
  }, 1000);
});

async function loadProducts() {
  const query = `
    {
      products(first: 10) {
        edges {
          node {
            id
            title
            description
            images(first:1) { edges { node { src } } }
            variants(first:1) { edges { node { id priceV2 { amount currencyCode } } } }
          }
        }
      }
    }
  `;
  const response = await fetch(`https://${storeDomain}/api/2025-10/graphql.json`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-Shopify-Storefront-Access-Token': storefrontToken
    },
    body: JSON.stringify({ query })
  });

  const data = await response.json();
  const products = data.data.products.edges;

  products.forEach(({ node }) => {
    const productDiv = document.createElement('div');
    productDiv.classList.add('product-card');
    productDiv.innerHTML = `
      <img src="${node.images.edges[0]?.node.src || ''}" />
      <h3>${node.title}</h3>
      <p>${node.description}</p>
      <p>Price: ${node.variants.edges[0].node.priceV2.amount} ${node.variants.edges[0].node.priceV2.currencyCode}</p>
      <button onclick="addToCart('${node.variants.edges[0].node.id}','${node.title}',${node.variants.edges[0].node.priceV2.amount})">Add to Cart</button>
    `;
    productsContainer.appendChild(productDiv);
  });
}

function addToCart(variantId, title, price) {
  const existing = cart.find(item => item.variantId === variantId);
  if (existing) {
    existing.quantity += 1;
  } else {
    cart.push({ variantId, title, price, quantity: 1 });
  }
  renderCart();
}

function removeFromCart(variantId) {
  cart = cart.filter(item => item.variantId !== variantId);
  renderCart();
}

function renderCart() {
  cartContainer.innerHTML = '';
  let total = 0;
  cart.forEach(item => {
    total += item.price * item.quantity;
    const div = document.createElement('div');
    div.classList.add('cart-item');
    div.innerHTML = `
      ${item.title} x ${item.quantity} - ${item.price * item.quantity} USD
      <button onclick="removeFromCart('${item.variantId}')">Remove</button>
    `;
    cartContainer.appendChild(div);
  });
  cartTotalEl.textContent = 'Total: ' + total.toFixed(2) + ' USD';
}

document.getElementById('checkoutBtn').addEventListener('click', async () => {
  if(cart.length === 0) return alert('Cart is empty!');
  
  const lineItems = cart.map(item => ({
    variantId: item.variantId,
    quantity: item.quantity
  }));

  const mutation = `
    mutation checkoutCreate($input: CheckoutCreateInput!) {
      checkoutCreate(input: $input) { checkout { webUrl } }
    }
  `;

  const response = await fetch(`https://${storeDomain}/api/2025-10/graphql.json`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-Shopify-Storefront-Access-Token': storefrontToken
    },
    body: JSON.stringify({ query: mutation, variables: { input: { lineItems } } })
  });

  const result = await response.json();
  const url = result.data.checkoutCreate.checkout.webUrl;
  window.open(url, '_blank');
});

  p.style.animationDuration = `${Math.random()*15+5}s`;
  particlesContainer.appendChild(p);
}
